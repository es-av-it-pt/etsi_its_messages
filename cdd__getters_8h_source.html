153 inline gm::PointStamped getUTMPosition(const ReferencePosition& reference_position, int& zone, bool& northp){
154 gm::PointStamped utm_point;
155 double latitude = getLatitude(reference_position.latitude);
156 double longitude = getLongitude(reference_position.longitude);
157 utm_point.point.z = getAltitude(reference_position.altitude);
158 try {
159 GeographicLib::UTMUPS::Forward(latitude, longitude, zone, northp, utm_point.point.x, utm_point.point.y);
160 std::string hemisphere;
161 if(northp) hemisphere="N";
162 else hemisphere="S";
163 utm_point.header.frame_id="utm_"+std::to_string(zone)+hemisphere;
164 } catch (GeographicLib::GeographicErr& e) {
165 throw std::invalid_argument(e.what());
166 }
167 return utm_point;
168 }

177 inline std::vector<bool> getBitString(const std::vector<uint8_t>& buffer, const int bits_unused) {
178 // bit string size
179 const int bits_per_byte = 8;
180 const int n_bytes = buffer.size();
181 const int n_bits = n_bytes * bits_per_byte;
182 std::vector<bool> bits;
183 bits.resize(n_bits - bits_unused, 0);
184
185 // loop over bytes in reverse order
186 for (int byte_idx = n_bytes - 1; byte_idx >= 0; byte_idx--) {
187
188 // loop over bits in a byte
189 for (int bit_idx_in_byte = 0; bit_idx_in_byte < bits_per_byte; bit_idx_in_byte++) {
190
191 // map bit index in byte to bit index in total bitstring
192 int bit_idx = (n_bytes - byte_idx - 1) * bits_per_byte + bit_idx_in_byte;
193 if (byte_idx == 0 && bit_idx >= n_bits - bits_unused) break;
194
195 // extract bit from bitstring and set output array entry appropriately
196 bool byte_has_true_bit = buffer[byte_idx] & (1 << bit_idx_in_byte);
197 if (byte_has_true_bit) bits[bit_idx] = 1;
198 }
199 }
200 return bits;
201 }
