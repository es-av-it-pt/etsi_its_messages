125inline std::vector<bool> getBitString(const std::vector<uint8_t>& buffer, const int bits_unused) {
126 // bit string size
127 const int bits_per_byte = 8;
128 const int n_bytes = buffer.size();
129 const int n_bits = n_bytes * bits_per_byte;
130 std::vector<bool> bits;
131 bits.resize(n_bits - bits_unused, 0);
132
133 // loop over bytes in reverse order
134 for (int byte_idx = n_bytes - 1; byte_idx >= 0; byte_idx--) {
135
136 // loop over bits in a byte
137 for (int bit_idx_in_byte = 0; bit_idx_in_byte < bits_per_byte; bit_idx_in_byte++) {
138
139 // map bit index in byte to bit index in total bitstring
140 int bit_idx = (n_bytes - byte_idx - 1) * bits_per_byte + bit_idx_in_byte;
141 if (byte_idx == 0 && bit_idx >= n_bits - bits_unused) break;
142
143 // extract bit from bitstring and set output array entry appropriately
144 bool byte_has_true_bit = buffer[byte_idx] & (1 << bit_idx_in_byte);
145 if (byte_has_true_bit) bits[bit_idx] = 1;
146 }
147 }
148 return bits;
149}
