194 inline void setExteriorLights(CAM& cam, const std::vector<bool>& exterior_lights){
195 if(etsi_its_cam_msgs::ExteriorLights::SIZE_BITS != exterior_lights.size()) {
196 throw std::invalid_argument("Vector has wrong size. (" + std::to_string(exterior_lights.size()) + " != " + std::to_string(etsi_its_cam_msgs::ExteriorLights::SIZE_BITS) + ")");
197 }
198 if(cam.cam.cam_parameters.low_frequency_container_is_present) {
199 if(cam.cam.cam_parameters.low_frequency_container.choice == etsi_its_cam_msgs::LowFrequencyContainer::CHOICE_BASIC_VEHICLE_CONTAINER_LOW_FREQUENCY) {
200 cdd::setExteriorLights(cam.cam.cam_parameters.low_frequency_container.basic_vehicle_container_low_frequency.exterior_lights, exterior_lights);
201 }
202 else {
203 throw std::invalid_argument("LowFrequencyContainer is not BASIC_VEHICLE_CONTAINER_LOW_FREQUENCY!");
204 }
205 }

207 throw std::invalid_argument("LowFrequencyContainer is not present!");
208 }
