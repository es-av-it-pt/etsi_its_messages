202template <typename T>
203inline void setBitString(T& bitstring, const std::vector<bool>& bits) {
204 // bit string size
205 const int bits_per_byte = 8;
206 const int n_bytes = (bits.size() - 1) / bits_per_byte + 1;
207 const int n_bits = n_bytes * bits_per_byte;
208
209 // init output
210 bitstring.bits_unused = n_bits - bits.size();
211 bitstring.value = std::vector<uint8_t>(n_bytes);
212
213 // loop over all bytes in reverse order
214 for (int byte_idx = n_bytes - 1; byte_idx >= 0; byte_idx--) {
215
216 // loop over bits in a byte
217 for (int bit_idx_in_byte = 0; bit_idx_in_byte < bits_per_byte; bit_idx_in_byte++) {
218
219 // map bit index in byte to bit index in total bitstring
220 int bit_idx = (n_bytes - byte_idx - 1) * bits_per_byte + bit_idx_in_byte;
221 if (byte_idx == 0 && bit_idx >= n_bits - bitstring.bits_unused) break;
222
223 // set bit in output bitstring appropriately
224 bitstring.value[byte_idx] |= bits[bit_idx] << bit_idx_in_byte;
225 }
226 }
227}
