117inline void setLongitudinalAccelerationValue(AccelerationValue& accel, const double value) {
118 int64_t accel_val = (int64_t)std::round(value*1e1);
119 if(accel_val>=AccelerationValue::MIN && accel_val<=AccelerationValue::MAX) accel.value = accel_val;
120 else if(accel_val<AccelerationValue::MIN) accel.value = AccelerationValue::MIN;
121 else if(accel_val>AccelerationValue::MAX) accel.value = AccelerationValue::MAX-1;
122}

175inline void setFromUTMPosition(ReferencePositionWithConfidence& reference_position, const gm::PointStamped& utm_position, const int zone, const bool northp)
176{
177 std::string required_frame_prefix = "utm_";
178 if(utm_position.header.frame_id.rfind(required_frame_prefix, 0) != 0)
179 {
180 throw std::invalid_argument("Frame-ID of UTM Position '"+utm_position.header.frame_id+"' does not start with required prefix '"+required_frame_prefix+"'!");
181 }
182 double latitude, longitude;
183 try {
184 GeographicLib::UTMUPS::Reverse(zone, northp, utm_position.point.x, utm_position.point.y, latitude, longitude);
185 } catch (GeographicLib::GeographicErr& e) {
186 throw std::invalid_argument(e.what());
187 }
188 setReferencePosition(reference_position, latitude, longitude, utm_position.point.z);
189}
