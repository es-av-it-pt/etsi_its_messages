332 inline void setFromUTMPosition(ReferencePosition& reference_position, const gm::PointStamped& utm_position, const int zone, const bool northp)
333 {
334 std::string required_frame_prefix = "utm_";
335 if(utm_position.header.frame_id.rfind(required_frame_prefix, 0) != 0)
336 {
337 throw std::invalid_argument("Frame-ID of UTM Position '"+utm_position.header.frame_id+"' does not start with required prefix '"+required_frame_prefix+"'!");
338 }
339 double latitude, longitude;
340 try {
341 GeographicLib::UTMUPS::Reverse(zone, northp, utm_position.point.x, utm_position.point.y, latitude, longitude);
342 } catch (GeographicLib::GeographicErr& e) {
343 throw std::invalid_argument(e.what());
344 }
345 setReferencePosition(reference_position, latitude, longitude, utm_position.point.z);
346 }

355 template <typename T>
356 inline void setBitString(T& bitstring, const std::vector<bool>& bits) {
357 // bit string size
358 const int bits_per_byte = 8;
359 const int n_bytes = (bits.size() - 1) / bits_per_byte + 1;
360 const int n_bits = n_bytes * bits_per_byte;
361
362 // init output
363 bitstring.bits_unused = n_bits - bits.size();
364 bitstring.value = std::vector<uint8_t>(n_bytes);
365
366 // loop over all bytes in reverse order
367 for (int byte_idx = n_bytes - 1; byte_idx >= 0; byte_idx--) {
368
369 // loop over bits in a byte
370 for (int bit_idx_in_byte = 0; bit_idx_in_byte < bits_per_byte; bit_idx_in_byte++) {
371
372 // map bit index in byte to bit index in total bitstring
373 int bit_idx = (n_bytes - byte_idx - 1) * bits_per_byte + bit_idx_in_byte;
374 if (byte_idx == 0 && bit_idx >= n_bits - bitstring.bits_unused) break;
375
376 // set bit in output bitstring appropriately
377 bitstring.value[byte_idx] |= bits[bit_idx] << bit_idx_in_byte;
378 }
379 }
380 }
